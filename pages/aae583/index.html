# 3. ORMжҹҘиҜў(13жқЎ) еҝ…зҹҘеҝ…дјҡ13жқЎ

#all()   иҺ·еҸ–жүҖжңүзҡ„ж•°жҚ®   вҖ”вҖ”гҖӢ QuerySet   еҜ№иұЎеҲ—иЎЁ
ret = models.Person.objects.all()

# get()  иҺ·еҸ–ж»Ўи¶іжқЎд»¶зҡ„дёҖдёӘж•°жҚ®   вҖ”вҖ”гҖӢ еҜ№иұЎ
#  иҺ·еҸ–дёҚеҲ°жҲ–иҖ…еӨҡдёӘйғҪжҠҘй”ҷ
ret = models.Person.objects.get(pk=1)

# filter()    иҺ·еҸ–ж»Ўи¶іжқЎд»¶зҡ„жүҖжңүж•°жҚ®   вҖ”вҖ”гҖӢ QuerySet   еҜ№иұЎеҲ—иЎЁ
ret = models.Person.objects.filter(pk=1)

# exclude()    иҺ·еҸ–дёҚж»Ўи¶іжқЎд»¶зҡ„жүҖжңүж•°жҚ®   вҖ”вҖ”гҖӢ QuerySet   еҜ№иұЎеҲ—иЎЁ
ret = models.Person.objects.exclude(pk=1)

# values()жӢҝеҲ°еҜ№иұЎжүҖжңүзҡ„еӯ—ж®өе’Ңеӯ—ж®өзҡ„еҖј QuerySet  [ {} ,{} ]
# values('еӯ—ж®ө')  жӢҝеҲ°еҜ№иұЎжҢҮе®ҡзҡ„еӯ—ж®өе’Ңеӯ—ж®өзҡ„еҖј QuerySet  [ {} ,{} ]
ret = models.Person.objects.values('pid','name')

# values_list()    жӢҝеҲ°еҜ№иұЎжүҖжңүзҡ„еӯ—ж®өзҡ„еҖј    QuerySet  [ () ,() ]
# values('еӯ—ж®ө')   жӢҝеҲ°еҜ№иұЎжҢҮе®ҡзҡ„еӯ—ж®өзҡ„еҖј      QuerySet  [ {} ,{} ]
ret = models.Person.objects.values_list('name','pid')

# order_by  жҺ’еәҸ   - йҷҚеәҸеҠ дёҖдёӘиҙҹеҸ·    вҖ”вҖ”гҖӢ QuerySet  [ () ,() ]
ret = models.Person.objects.all().order_by('age','-pid') #дј еӨҡдёӘеҸӮж•°

# reverse  еҸҚеҗ‘жҺ’еәҸ   еҸӘиғҪеҜ№е·Із»ҸжҺ’еәҸзҡ„QuerySetиҝӣиЎҢеҸҚиҪ¬
ret = models.Person.objects.all()
ret = models.Person.objects.all().reverse()

# distinct еҺ»йҮҚ  е®Ңе…ЁзӣёеҗҢзҡ„еҶ…е®№жүҚиғҪеҺ»йҮҚ
ret = models.Person.objects.values('age').distinct()

#  count()  и®Ўж•°
ret = models.Person.objects.all().count()

# first  еҸ–з¬¬дёҖе…ғзҙ    жІЎжңүе…ғзҙ  None
# last  еҸ–жңҖеҗҺдёҖе…ғзҙ    жІЎжңүе…ғзҙ  None
ret = models.Person.objects.filter(pk=1).values().first()

# exists жҹҘиҜўзҡ„ж•°жҚ®жҳҜеҗҰеӯҳеңЁ
ret = models.Person.objects.filter(pk=1000).exists()
